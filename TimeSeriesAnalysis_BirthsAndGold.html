#function to add time lags to the predictor variables in a new matrix
time_lag_matrix2 <- function (x, maxlag) {
    p = ncol(x)
    N = nrow(x) - maxlag - 1
#    x_new_num_col = maxlag * p + p #as a check
    x_new = NULL
    for (j in 1:p) {
      for (i in 1:(maxlag + 1)) {
        x_temp = x[(i):(N + i), j]
        x_new = cbind(x_new,as.vector(x_temp))
      }
    }
    return(x_new)
}
